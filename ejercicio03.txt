#include <iostream>
using namespace std;

float suma (float numero);
float factorial (float i);

float suma (float numero){
    float s = 0;
    for (float i = 1; i <= numero; i++){
        s = s + (factorial(i)/(2 * i));
    }
    return s;
}

float factorial (float i){
    float f = 1;
    for (float j = 2; j <= i;j++){
        f = f * j;
    }
    return f;
}
